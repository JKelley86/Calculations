const wordList = ["apple", "bread", "crane", "drive", "eagle", "flame", "grape"]; let currentWord = ""; let currentGuess = ""; let rowIndex = 0; const maxGuesses = 6; document.getElementById('start-button').addEventListener('click', startGame); document.getElementById('reset-button').addEventListener('click', resetGame); function startGame() { currentWord = wordList[Math.floor(Math.random() * wordList.length)]; currentGuess = ""; rowIndex = 0; document.getElementById('message').textContent = ""; generateGrid(); generateKeyboard(); document.getElementById('start-button').style.display = 'none'; document.getElementById('reset-button').style.display = 'none'; } function resetGame() { startGame(); } function generateGrid() { const grid = document.getElementById('grid'); grid.innerHTML = ""; for (let i = 0; i < maxGuesses * 5; i++) { const cell = document.createElement('div'); cell.className = 'grid-cell'; cell.setAttribute('data-index', i); grid.appendChild(cell); } } function generateKeyboard() { const keyboardLayout = [ 'q', 'w', 'e', 'r', 't', 'y', 'u', 'i', 'o', 'p', 'a', 's', 'd', 'f', 'g', 'h', 'j', 'k', 'l', 'z', 'x', 'c', 'v', 'b', 'n', 'm' ]; const keyboard = document.getElementById('keyboard'); keyboard.innerHTML = ""; keyboardLayout.forEach(key => { const button = document.createElement('button'); button.className = 'key'; button.textContent = key; button.addEventListener('click', () => handleKeyPress(key)); keyboard.appendChild(button); }); const enterKey = document.createElement('button'); enterKey.className = 'key large'; enterKey.textContent = 'Enter'; enterKey.addEventListener('click', handleEnter); keyboard.appendChild(enterKey); const deleteKey = document.createElement('button'); deleteKey.className = 'key large'; deleteKey.textContent = 'Del'; deleteKey.addEventListener('click', handleDelete); keyboard.appendChild(deleteKey); } function handleKeyPress(key) { if (currentGuess.length < 5) { currentGuess += key; updateGrid(); } } function handleEnter() { if (currentGuess.length === 5) { checkGuess(); if (currentGuess === currentWord || rowIndex === maxGuesses - 1) { document.getElementById('message').textContent = currentGuess === currentWord ? "Congratulations! You've guessed the word!" : `Game Over! The word was ${currentWord}.`; document.getElementById('reset-button').style.display = 'block'; } currentGuess = ""; rowIndex++; } } function handleDelete() { if (currentGuess.length > 0) { currentGuess = currentGuess.slice(0, -1); updateGrid(); } } function updateGrid() { const start = rowIndex * 5; for (let i = 0; i < 5; i++) { const cell = document.querySelector(`.grid-cell[data-index='${start + i}']`); cell.textContent = currentGuess[i] || ''; } } function checkGuess() { const start = rowIndex * 5; const guess = currentGuess.split(''); const correctWord = currentWord.split(''); guess.forEach((letter, index) => { const cell = document.querySelector(`.grid-cell[data-index='${start + index}']`); if (correctWord[index] === letter) { cell.classList.add('correct'); updateKeyboardKeyColor(letter, 'correct'); } else if (correctWord.includes(letter)) { cell.classList.add('present'); updateKeyboardKeyColor(letter, 'present'); } else { cell.classList.add('absent'); updateKeyboardKeyColor(letter, 'absent'); } }); } function updateKeyboardKeyColor(letter, status) { const key = [...document.querySelectorAll('.key')].find(k => k.textContent === letter); if (key) { const currentStatus = key.className.split(' ').pop(); if (currentStatus === 'absent' && status !== 'absent') { key.className = `key ${status}`; } else if (currentStatus === 'present' && status === 'correct') { key.className = `key ${status}`; } } }
